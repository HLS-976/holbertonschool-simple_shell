char *strdup(const char *s) {
    size_t len = strlen(s) + 1; // Longueur de la chaîne + 1 pour le caractère nul
    char *new_str = (char *)malloc(len); // Allocation de mémoire

    if (new_str == NULL) {
        return NULL; // Échec de l'allocation de mémoire
    }

    return (char *)memcpy(new_str, s, len); // Copie de la chaîne
}
